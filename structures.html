1. Create structure of employee which there is emp.ID, name, designation of 50 employees from user, print details in tabular form ?

Code:

#include <stdio.h>
struct employee
{
    int id;
    char name[20];
    char design[25];
};

int main()
{
    struct employee emp[50];

    printf("Enter details of 50 employees:\n");

    for (int i = 0; i < 50; i++)
    {

        printf("\nEmployee %d:\n", i + 1);
        printf("Enter ID: ");
        scanf("%d", &emp[i].id);
        fflush(stdin);

        printf("Enter Name: ");
        scanf("%s", emp[i].name);
        fflush(stdin);
        printf("Enter Designation: ");
        scanf("%s", emp[i].design);
    }

    printf("\n---------------------------------------------------------------\n");
    printf("| %-5s | %-25s | %-20s |\n", "ID", "Name", "Designation");
    printf("---------------------------------------------------------------\n");

    for (int i = 0; i < 50; i++)
    {
        printf("| %-5d | %-25s | %-20s |\n", emp[i].id, emp[i].name, emp[i].design);
    }

    printf("---------------------------------------------------------------\n");

    return 0;
}

4. Enter student details like roll name, fee, DOB and print all the detail by ascending order of roll number ?

Code:

#include <stdio.h>
struct student
{
    int roll;
    char name[20];
    int fee;
    char dob[20];
};

int main()
{
    struct student stu[5], temp;

    printf("Enter details of 5 students:\n");

    for (int i = 0; i < 5; i++)
    {

        printf("\nStudent %d:\n", i + 1);
        printf("Enterc your roll number: ");
        scanf("%d", &stu[i].roll);
        fflush(stdin);

        printf("Enter your Name: ");
        scanf("%s", stu[i].name);
        fflush(stdin);

        printf("Enter the fees: ");
        scanf("%d", stu[i].fee);
        fflush(stdin);

        printf("Enter the DOB: ");
        scanf("%s", stu[i].dob);
        fflush(stdin);
    }

    for (int i = 0; i < 5 - 1; i++)
    {
        for (int j = i + 1; j < 5; j++)
        {
            if (stu[i].roll > stu[j].roll)
            {
                temp = stu[i];
                stu[i] = stu[j];
                stu[j] = temp;
            }
        }
    }

    printf("\n-------------------------------------------------------------\n");
    printf("| %-10s | %-15s | %-10s | %-12s |\n", "Roll No", "Name", "Fee", "DOB");
    printf("-------------------------------------------------------------\n");

    for (int i = 0; i < 5; i++)
    {
        printf("| %-10d | %-15s | %-10d | %-12s |\n",
               stu[i].roll, stu[i].name, stu[i].fee, stu[i].dob);
    }

    printf("-------------------------------------------------------------\n");

    return 0;
}
