    int i,f=1;

        f=f*i;
    printf("Factorial = %d\n", f);
}
void reverse() {
    int r=0,temp=n;
    while(temp!=0) {
        r =r*10+(temp%10);
        temp/=10;
    }
    printf("Reverse = %d\n", r);
}
void palindrome() {
    int temp=n,r=0; 
    while(temp!=0) {
        r = r*10+(temp%10);
        temp/=10;
    }
    if(r==n)
        printf("Palindrome = Yes\n");
    else
        printf("Palindrome = No\n");
}
void prime() {
    int i, c=0; 
    for(i =1;i<=n;i++) {
        if(n %i==0)
            c++;
    }
    if(c == 2)
        printf("Prime = Yes\n");
    else
        printf("Prime = No\n");
}
void armstrong() {
    int temp =n, sum =0, r; 
    while(temp !=0) {
        r = temp%10;
        sum +=r*r*r;
        temp/=10;
    }
    if(sum == n)
        printf("Armstrong = Yes\n");
    else
        printf("Armstrong = No\n");
}

#include <stdio.h>
void swap(int *x,int *y){
    int z;
    z=*x;
    *x=*y;
    *y=z;
}
void og(int a,int b){
    printf("original value of \na=%d\nb=%d\n ",a,b);
}

int a,b;
