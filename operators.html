1.Write a program to show that example of pre-increment ?

Code:

#include <stdio.h>
void main() {
    int a = 5;
    int b = ++a;
    printf("a=%d, b=%d", a, b);
}

Output:

a=6, b=6

11. Enter marks of students and calculate average of student’s marks ?

Code:

#include <stdio.h>
void main() {
    int n, i, marks, sum=0;
    printf("Enter number of students: ");
    scanf("%d", &n);
    for(i=1;i<=n;i++){
        printf("Enter marks of student %d: ", i);
        scanf("%d",&marks);
        sum += marks;
    }
    printf("Average marks: %.2f", (float)sum/n);
}

Output:

Enter number of students: 3
Enter marks of student 1: 77
Enter marks of student 2: 58
Enter marks of student 3: 98
Average marks: 77.67

12. Calculate simple interest = (P x R x T)/ 100

Code:

#include <stdio.h>
void main() {
    float P, R, T, SI;
    printf("Enter Principal, Rate, Time: ");
    scanf("%f %f %f",&P,&R,&T);
    SI = (P*R*T)/100;
    printf("Simple Interest: %.2f", SI);
}

Output:

Enter Principal, Rate, Time: 2 3 4
Simple Interest: 0.24
